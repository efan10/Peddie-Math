\section{Question 2}
Let $\alpha,\beta,\gamma$ be the roots of the cubic equation $x^3-3x^2+3x+7=0$. Given that $\omega$ is a non-real complex cube root of unity, determine the numerical value of the expression:
\[
\frac{\alpha -1}{\beta -1}+\frac{\beta -1}{\gamma -1}+\frac{\gamma -1}{\alpha -1}
\]
\begin{gather*}
    x^3-3x^2+3x+7=0\\
    (x-1)^3=-8\\
    \frac{1-x}{2}=1, \frac{1+i\sqrt3}{2}, \frac{1-i\sqrt3}{2}\\
    x-1=-2, -1-i\sqrt{3}, -1+i\sqrt{3}\\
    \frac{\alpha -1}{\beta -1}+\frac{\beta -1}{\gamma -1}+\frac{\gamma -1}{\alpha -1}\\
    \frac{-2}{-1-i\sqrt{3}}+\frac{-1-i\sqrt{3}}{-1+i\sqrt{3}}+\frac{-1+i\sqrt{3}}{-2}\\
    \frac{2-i2\sqrt{3}}{2}+\frac{1-3+i2\sqrt{3}}{2}+\frac{1-i\sqrt{3}}{2}\\
    \boxed{\frac{1-i\sqrt{3}}{2}}
\end{gather*}

\section{Question 4}
Find the common roots of $x^{12}-1=0$ and $x^4+x^2+1=0$.
\begin{gather*}
    x^{12}=1 \implies x=e^{i\frac{k\pi}{12}},\ k\in\mathbb{Z}\\
    x^4+x^2+1 \implies x^2=\frac{-1\pm i\sqrt{3}}{2}=e^{i\frac{2\pi}{3}} \text{ or } e^{i\frac{4\pi}{3}}\\
    \boxed{x=e^{i\frac{\pi}{3}},e^{i\frac{4\pi}{3}},e^{i\frac{2\pi}{3}}, \, \text{or}\  e^{i\frac{5\pi}{3}}}
\end{gather*}

\section{Question 11}
Let $\dfrac{1}{a+\omega }+\dfrac{1}{b+\omega }+\dfrac{1}{c+\omega }+\dfrac{1}{d+\omega }=\dfrac{1}{\omega}$, where $a,b,c,d \in \mathbb{R}$ and $\omega$ is a complex cube root of unity. 
Find the value of
\[
\sum\frac{1}{a^2-a+1}
\]
\begin{gather*}
    \omega = \frac{-1+i\sqrt{3}}{2}, \ \omega^2=\frac{-1-i\sqrt{3}}{2}\\
    \frac{1}{a^2-a+1}=\frac{1}{(a+\omega)(a+\omega^2)}=\frac{1}{\omega-\omega^2}(\frac{1}{a+\omega}-\frac{1}{a+\omega^2})\\
    \sum\frac{1}{a^2-a+1}=\frac{1}{\omega-\omega^2}(\sum \frac{1}{a+\omega}-\sum \frac{1}{a+\omega^2})\\
    \frac{1}{\omega-\omega^2}(\frac{1}{\omega}-\frac{1}{\omega^2})=\frac{1}{\omega-\omega^2}(\frac{\omega^2-\omega}{\omega  \omega^2})=\frac{-1}{1}\\
    \boxed{-1}
\end{gather*}
