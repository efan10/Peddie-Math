\problem{1}
Show that each of the following functions is differentiable at $x = 0$. \vem

\subproblem{a} $f(x)=\cos \abs x$
\begin{gather*}
    f(x)=\begin{cases}
        \cos x &x \ge 0\\
        \cos x &x<0
    \end{cases}\\
    f'(x)=\begin{cases}
        - \sin x & x>0\\
        - \sin x & x<0
    \end{cases} \\
    f'(0^-)=-\sin 0=0\\
    f'(0^+)= -\sin 0 =0\\
    f'(0^-)=f'(0^+)=0 \implies f'(0)=0\\
    \boxed{f(x) \text{ is differentiable at $x=0$}}
\end{gather*}

\subproblem{c} $f(x)=|x^3|$
\begin{gather*}
    f(x)=\begin{cases}
        x^3 &x\ge 0\\
        -x^3 &x<0
    \end{cases}\\
    f'(x)=\begin{cases}
        3x^2 &x>0\\
        -3x^2 &x<0
    \end{cases}\\
    f'(0^-)=3\cdot0^2=0\\
    f'(0^+)=-3\cdot 0^2=0\\
    f'(0^-)=f'(0^+)=0 \implies f'(0)=0\\
    \boxed{f(x) \text{ is differentiable at $x=0$}}
\end{gather*}

\problem{3}
Let $f(x)=\begin{cases}
    ax(x-1)+b & \text{if }x<1\\
    x-1 &\text{if }1\le x \le 3\\
    px^2+qx+2 &\text{if }x>3
\end{cases}$ \vem \\
Find the values of $a,b,p,$ and $q$ that satisfy the following set of conditions.
\begin{enumerate}
    \item $f(x)$ is continuous for all $x$.
    \item $f'(1)$ doesn't exist.
    \item $f'(x)$ is continuous at $x=3$.
\end{enumerate}
\begin{enumerate}
    \item \begin{gather*}
    f(1^-)=a\cdot1\cdot(1-1)+b=b\\
    f(1^+)=1-1=0\\
    f(1^-)=f(1^+)=0 \implies b=0\\
    f(3^-)=3-1=2\\
    f(3^+)=9p+3q+2\\
    f(3^-)=f(3^+) \implies 9p+3q=0\\
\end{gather*}
    \item \begin{gather*}
        f'(x)=\begin{cases}
        2ax-a &x<1\\
        1 & 1 < x < 3\\
        2px+q &x>3
    \end{cases}\\
    f'(1^-)=2a-a=a\\
    f'(1^+)=1\\
    f'(1^-)\ne f'(1^+) \implies a\ne 1
    \end{gather*}
    \item \begin{gather*}
        f'(3^-)=1\\
        f'(3^+)=2p+q\\
        f'(3^-)=f'(3^+)\implies 2p+q=1
    \end{gather*}
\end{enumerate}
\begin{gather*}
    \begin{cases}
        9p+3q=0 \\
        6p+q=1\\
    \end{cases} \implies p=\frac{1}{3}, q=-1\\
    \boxed{a\ne 1,b=0,p=\frac{1}{3}, q=-1}
\end{gather*}

\problem{4}
 Sketch the function by hand. Identify where the derivative does not exist and explain the specific reasons at those points. \vem

 \subproblem{a} $f(x)=\sin |x|$
 \begin{gather*}
    f(x)=\begin{cases}
        \sin x &x \ge 0\\
        \sin -x & x<0
    \end{cases}\\
    f'(x)=\begin{cases}
        \cos x &x >0\\
        -\cos x &x<0
    \end{cases}\\
     f'(0^+)=\cos 0 = 1\\
     f'(0^-)=- \cos 0 =-1\\
     f'(0^+) \ne f'(0^-) \implies f'(0) \dne\\
     \boxed{f(x) $ has corner non-differentiability at $ x=0}
 \end{gather*}

 \subproblem{c} $f(x)= \arcsin |\sin x|$
 \begin{gather*}
     f(x)=\begin{cases}
         \arcsin \sin x & x\in [2k\pi, (2k+1)\pi], k\in \mathbb{Z}\\
         \arcsin -\sin x & x\in [(2k+1)\pi, (2k+2)\pi], k\in \mathbb{Z}
     \end{cases}\\
     f(x)=\begin{cases}
         x &x\in [0,\frac{\pi}{2}]\\
         \pi-x &x\in [\frac{\pi}{2},\pi]
     \end{cases}\\
     f'(x)=\begin{cases}
         1 &x\in (0,\frac{\pi}{2})\\
         -1 &x\in (\frac{\pi}{2},\pi)
     \end{cases}\\
     f'(\frac{\pi}{2}^+)=-1, f'(\frac{\pi}{2}^-)=1\\
     f'(\frac{\pi}{2}^+)\ne f'(\frac{\pi}{2}^-) \implies f'(\frac{\pi}{2}) \dne
     \end{gather*}
Similarly, the corner non-differentiability applies to all $x=\dfrac{k\pi}{2}, k\in \mathbb{Z}$.
\[
\boxed{f(x) $ has corner non-differentiability at $ x=\dfrac{k\pi}{2}, k\in \mathbb{Z}}
\]

5. Discuss the differentiability of \vem
\subproblem{a} $f(x)=|x|+|x+1|$
\begin{gather*}
    f(x)=\begin{cases}
        -2x-1 &x<-1\\
        1 & -1\le x < 0\\
        2x+1 &x>0
    \end{cases}\\
    f'(x)=\begin{cases}
        -2 &x<-1\\
        0 & -1\ <x < 0\\
        2 &x>0
    \end{cases}\\
    f'(-1^-)=-2, f'(-1^+)=0\\
    f'(-1^-) \ne f'(-1^+) \implies f'(-1) \dne\\
    f'(0^-)=0, f'(0^+)=2\\
    f'(0^-) \ne f'(0^+) \implies f'(0) \dne\\
    \boxed{f(x) $ has corner non-differentiability at $ x=-1,0}
\end{gather*}

\subproblem{b} $f(x)=\floor{x}+|1-x|, x\in (-1,3)$
\begin{gather*}
    f(x)=\begin{cases}
        -x &x\in (-1,0)\\
        1-x &x\in [0,1)\\
        x &x \in [1,2)\\
        x+1 & x \in [2,3)
    \end{cases}\\
    f(0^-)=0, f(0^+)=1\\
    f(0^-) \ne f(0^+) \implies \limit 0 f(x) \dne 
\end{gather*}
Similarly, $f(x)$ is also discontinuous and thus non-differentiable on $x=1,2$.
\begin{equation*}
    \boxed{f(x) $ is discontinuous on $ x=0,1,2}
\end{equation*}

\subproblem{c} $f(x)=(x^2-1)|x^2-x-2|+\sin |x|$
\begin{gather*}
    f(x)=(x^2-1)|(x-2)(x+1)|+\sin |x|\\
    f(x)=\begin{cases}
        (x^2-1)(x^2-x-2)+\sin -x & x<-1\\
        -(x^2-1)(x^2-x-2)+\sin -x & -1 \le x<0\\
        -(x^2-1)(x^2-x-2)+\sin x & 0 \le x<2\\
        (x^2-1)(x^2-x-2)+\sin x & x \ge 2
    \end{cases}\\
    f'(x)=\begin{cases}
        4x^3-3x^2-6x+1- \cos x & x<-1\\
        -4x^3+3x^2+6x-1 -\cos x & -1 < x<0\\
        -4x^3+3x^2+6x-1+\cos x & 0 < x<2\\
        4x^3-3x^2-6x+1+\cos x & x > 2
    \end{cases}\\
    f'(0^-)=-1-\cos 0=-2\\
    f'(0^+)=-1+\cos 0 =0\\
    f'(0^-)\ne f'(0^+) \implies f'(0) \dne\\
    \boxed{f(x) $ is not differentiable at $ x=0}
\end{gather*}
 
\subproblem{d} $f(x)=|x|\sin x+|x-2| \sgn (x-2)+|x-3|$
\begin{gather*}
    f(x)=\begin{cases}
        -x \sin x+1 & x<0\\
        x \sin x+1 & 0 \le x <3\\
        x \sin x+2x-5 & x \ge 3
    \end{cases}\\
    f'(x)=\begin{cases}
        -x \cos x- \sin x & x<0\\
        x \cos x + \sin x & 0 < x <3\\
        x \cos x + \sin x+2 & x > 3
    \end{cases}\\
    f'(3^-)=3 \cos 3+ \sin 3\\
    f'(3^+)=3 \cos 3+ \sin 3 + 2\\
    f'(3^-) \ne f'(3^+) \implies f'(3) \dne\\
    \boxed{f(x) $ has corner non-differentiability at $ x=3}
\end{gather*}

\problem{7}
Let $f(x)=\begin{cases}
    (x-1)^2\sin \dfrac{1}{x-1}-|x| &\text{if }x\ne1\\
    -1 &\text{if }x=1
\end{cases}$ be a real-valued function. Find where $f(x)$ is not differentiable.
\begin{gather*}
    f(x)=\begin{cases}
    (x-1)^2\sin \dfrac{1}{x-1}+x &x <0\\
    (x-1)^2\sin \dfrac{1}{x-1}-x &0 \le x, x\ne 1\\
    -1 &x=1
    \end{cases}\\
    f'(x)=\begin{cases}
    -\cos \dfrac{1}{x-1}+\sin \dfrac{1}{x-1}(2x-2)+1 &x <0\\
    -\cos \dfrac{1}{x-1}+\sin \dfrac{1}{x-1}(2x-2)-1 &0 < x, x\ne 1\\
    -1 &x=1
    \end{cases}\\
    f'(0^-)=-\cos \frac{1}{0-1}+\sin \frac{1}{0-1}(2\cdot0-2)+1=-\cos 1 +2 \sin 1+1\\
    f'(0^+)=-\cos \frac{1}{0-1}+\sin \frac{1}{0-1}(2\cdot0-2)+1=-\cos 1 +2 \sin 1-1\\
    f'(0^-) \ne f'(0^+) \implies f'(0) \dne \\
    \boxed{f(x) $ is not differentiable at $ x=0}
\end{gather*}
