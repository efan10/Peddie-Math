\problem{1}
Find $\dydx:$ \vem 

\subproblem{a} $\sqrt{x}+\sqrt{y}=4$
\begin{gather*}
    x^\frac{1}{2}+y^\frac{1}{2}=4\\
    \frac{1}{2\sqrt{x}}+\frac{y'}{2\sqrt{y}}=0\\
    \boxed{\dydx =-\sqrt{\frac{y}{x}}}
\end{gather*}

\subproblem{b} $x^3+y^3=3xy$
\begin{gather*}
    3x^2+3y^2y'=3(y+xy')\\
    x^2-y=y'(x-y^2)\\
    \boxed{\dydx =\frac{x^2-y}{x-y^2}}
\end{gather*}

\subproblem{c} $xy+y^2=\tan x+y$\\
\begin{gather*}
    y+xy'+2yy'=\sec^2x+y'\\
    (x+2y-1)y'=\sec^2 x-y\\
    \boxed{\dydx = \frac{\sec^2 x-y}{x+2y-1}}
\end{gather*}

\subproblem{d} $\sec (x+y)=xy$
\begin{gather*}
    \sec (x+y)\tan (x+y)(1+y')=y+xy'\\
    \boxed{\dydx = \frac{y-\sec (x+y)\tan (x+y)}{\sec (x+y)\tan (x+y)-x}}
\end{gather*}

\problem{2}
Find: \vem 

\subproblem{a} $y'(0)$ if $4xe^{xy}=y+5\sin^2 x$.
\begin{gather*}
    4e^{xy}+4x e^{xy} (y+xy')=y'+10\sin x \cos x\\
    y'(4x^2e^{xy}-1)=10\sin x\cos x-4e^{xy}-4xye^{xy}\\
    y'=\frac{10\sin x\cos x-4e^{xy}-4xye^{xy}}{4x^2e^{xy}-1}\\
    x=0 \implies y=0\\
    y'(0)=\frac{10\sin 0\cos 0-4e^{0 \cdot 0}-4\cdot 0\cdot 0 e^{0\cdot 0}}{4\cdot 0^2e^{0 \cdot 0}-1}\\
    \boxed{y'(0)=4}
\end{gather*}

\subproblem{b} $y'(e)$ if $\ln (\ln x-\ln y)=e^{x^2y}(1-\ln x)$.
\begin{gather*}
    \ln (\ln\frac{x}{y})=e^{x^2y}-e^{x^2y} \ln x\\
    \frac{1}{\ln \frac{x}{y}}\cdot \frac{y-xy'}{xy}=e^{x^2y}(2xy+y'x^2)(1-\ln x)-\frac{1}{x}e^{x^2y}\\
    \frac{1}{x \ln \frac{x}{y}}+e^{x^2y}(\frac{1}{x}-2xy(1-\ln x))=y'(\frac{1}{y \ln x}+e^{x^2y}x^2(1- \ln x))\\
    y'=\frac{\frac{1}{x \ln \frac{x}{y}}+e^{x^2y}(\frac{1}{x}-2xy(1-\ln x))}{\frac{1}{y \ln x}+e^{x^2y}x^2(1- \ln x)}\\
    x=e \implies y=1\\
    y'(e)=\frac{\frac{1}{e}+e^{e^2-1}}{1}\\
    \boxed{y'(e)=\frac{e^{e^2}+1}{e}}
\end{gather*}

\problem{3}
Find the derivative $\dydx$: \vem

\subproblem{a} $y=\dfrac{\sqrt{x}(x+4)^\frac{3}{2}}{(4x-3)^\frac{4}{3}}$
\begin{gather*}
    \ln y = \ln \sqrt{x} +\ln (x+4)^\frac{3}{2}-\ln (4x-3)^{\frac{4}{3}}\\
    \frac{y'}{y}= \frac{1}{\sqrt{x}}\cdot\frac{1}{2\sqrt{x}}+\frac{\frac{3}{2}(x+4)^\frac{1}{2}}{(x+4)^\frac{3}{2}}-\frac{\frac{4}{3}(4x-3)^\frac{1}{3}4}{(4x-3)^\frac{4}{3}}\\
    y'=y(\frac{1}{2x}+\frac{3}{2(x+4)}-\frac{16}{3(4x-3)})\\
    \boxed{\dydx=\dfrac{\sqrt{x}(x+4)^\frac{3}{2}}{(4x-3)^\frac{4}{3}}(\frac{1}{2x}+\frac{3}{2(x+4)}-\frac{16}{3(4x-3)})}
\end{gather*}

\subproblem{b} $y=(\ln x)^{\cos x}$
\begin{gather*}
    \ln y =\cos x \ln (\ln x)\\
    \frac{y'}{y}=-\sin x\ln (\ln x)+\cos x \frac{1}{x \ln x}\\
    \boxed{\dydx =(\ln x)^{\cos x}(\cos x \frac{1}{x \ln x}-\sin x\ln (\ln x))}
\end{gather*}

\subproblem{c} $xy=e^{x-y}$
\begin{gather*}
    \ln xy=x-y\\
    \frac{1}{xy}(y+xy')=1-y'\\
    y+xy'=xy-xyy'\\
    y'(x+xy)=xy-y\\
    \boxed{\dydx =\frac{xy-y}{x+xy}}
\end{gather*}

\subproblem{d} $y^x=x^y$
\begin{gather*}
    \ln y^x=\ln x^y\\
    x\ln y=y\ln x\\
    \ln y +\frac{xy'}{y}=y'\ln x+\frac{y}{x}\\
    y'(\ln x-\frac{x}{y})=\ln y-\frac{y}{x}\\
    \boxed{\dydx = \frac{\ln y-\frac{y}{x}}{\ln x-\frac{x}{y}}}
\end{gather*}

\subproblem{e} $y=x^x$
\begin{gather*}
    \ln y=x \ln x\\
    \frac{y'}{y} =\ln x+1\\
    \boxed{\dydx = x^x(\ln x +1)}
\end{gather*}

\subproblem{f} $y=(\tan x)^{(\tan x)^{\tan x}}$
\begin{gather*}
    \ln y=(\tan x)^{\tan x}\ln \tan x\\
    \ln \ln y =\ln \ln \tan x+ \tan x \ln \tan x\\
    \frac{1}{\ln y} \cdot \frac{y'}{y}=\frac{1}{\ln \tan x}\cdot \frac{\sec ^2 x}{\tan x}+\sec^2
    x \cdot \ln \tan x +\tan x\frac{\sec^2x}{\tan x}\\
    y'=\sec ^2 x(\frac{1}{\tan x\ln \tan x}+\ln \tan x+1)y\ln y\\
    \boxed{\dydx = \sec ^2 x(\tan x)^{(\tan x)^{\tan x}}(\tan x)^{\tan x}\ln \tan x(\frac{1}{\tan x\ln \tan x}+\ln \tan x+1)}
\end{gather*}

\problem{4}
If $x^my^n=(x+y)^{m+n}$, prove that $\dydx=\dfrac{y}{x}$.
\begin{gather*}
    \ln x^my^n=\ln (x+y)^{m+n}\\
    m\ln x+n\ln y=(m+n)\ln (x+y)\\
    \frac{m}{x}+\frac{ny'}{y}=\frac{(m+n)}{(x+y)}\cdot (1+y')\\
    y'(\frac{m+n}{x+y}-\frac{n}{y})=\frac{m}{x}-\frac{m+n}{x+y}\\
    \dydx =\frac{\frac{mx+my-mx-nx}{x(x+y)}}{\frac{my+ny-nx-ny}{y(x+y)}}\\
    \boxed{\dydx = \frac{y}{x}}
\end{gather*}

\problem{5}
Find $y'(0)$ if $y=\dfrac{\sqrt[3]{1+3x}\sqrt[4]{1+4x}\sqrt[5]{1+5x}}{\sqrt[7]{1+7x}\sqrt[8]{1+8x}}$.
\begin{gather*}
    \ln y = \ln (\sqrt[3]{1+3x}\sqrt[4]{1+4x}\sqrt[5]{1+5x})-\ln (\sqrt[7]{1+7x}\sqrt[8]{1+8x})\\
    \frac{y'}{y}=\frac{1}{1+3x}+\frac{1}{1+4x}+\frac{1}{1+5x}-\frac{1}{1+7x}-\frac{1}{1+8x}\\
    x=0 \implies y=1\\
    y'(0)=\frac{1}{1}+\frac{1}{1}+\frac{1}{1}-\frac{1}{1}-\frac{1}{1}\\
    \boxed{y'(0)=1}
\end{gather*}
